function kasaHareketKaydet(tur, donemKey) {

const id = 'kh_' + (++window.kasaHarIdSay);

const tutar2Say = parseFloat(g('kH_tutar2'))||0;

const sym = {TL:'₺',USD:'$',EUR:'€'}[kasaDoviz]||'';

let hareket = {
id, tur, tarih: g('kH_tarih'),
cinsi: tur==='avans' ? (g('kH_avansType')==='borc'?'Borç avans':'Hizmet avansı') : g('kH_cinsi'),
kasaDoviz,
tutar: sym + new Intl.NumberFormat('tr-TR').format(tutarSay),

aciklama: g('kH_aciklama'),
avansKisi: g('kH_avansKisi'),
avansType: g('kH_avansType'),
avansGeriDondu: false,
};

// Döviz bozdurma — özel kayıt

function pazarBakiyeUSD(bakiye, doviz, eurUsd, gbpUsd, tryKur) {
if(doviz==='USD') return bakiye;
if(doviz==='GBP') return bakiye * gbpUsd;
if(doviz==='EUR') return bakiye * eurUsd;
if(doviz==='TRY') return bakiye / tryKur;
return bakiye;
}

function renderPazaryeri() {
const sel = document.getElementById('globalDonem');
if(!sel) return;
const idx = parseInt(sel.value);
const d = DONEMLER[idx];
const donemKey = d.y+'-'+d.m;
const tryKur = window.getKur(d.y,d.m);
const eurUsd = window.getEurUsd(d.y,d.m);
const gbpUsd = window.getGbpUsd(d.y,d.m);
const veri = window.PAZAR_VERI[donemKey] || JSON.parse(JSON.stringify(window.PAZAR_VERI['2026-4']||[]));
const fU = v => '$'+new Intl.NumberFormat('en-US',{minimumFractionDigits:0,maximumFractionDigits:0}).format(Math.round(v));
const symMap = {USD:'$',GBP:'£',EUR:'€',TRY:'₺'};

let toplam=0;
const rows = veri.map((p,i) => {
const usd = pazarBakiyeUSD(p.bakiye, p.doviz, eurUsd, gbpUsd, tryKur);
toplam += usd;
const sym = symMap[p.doviz]||'';
const bakiyeGoster = p.bakiye>0 ? sym+new Intl.NumberFormat('tr-TR').format(p.bakiye) : '—';
const usdGoster = usd>0 ? fU(usd) : '—';

+'<td><span class="badge '+dovizBadge+'">'+p.doviz+'</span></td>'
